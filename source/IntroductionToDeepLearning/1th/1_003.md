## 一、 手动模拟广播：manual_broadcast(a, b)
底层硬核逻辑：

   1. 右对齐与维度补齐：比较两者的 shape，在较短的数组左边补 1（通过 reshape 隐式增加中括号）。
   2. 跨步欺骗与重复：遍历每个轴，如果某个轴大小不匹配且其中一个是 1，则使用 np.repeat（底层相当于将 C 步长 Strides 设为 0 并开辟新内存硬拷贝）将该轴复制到目标尺寸。

import numpy as np
def manual_broadcast(a, b):
    # 1. 自动向左补齐维度 (例如: a=(3,), b=(4,3) -> a_shape=(1,3))
    a_shape = list(a.shape)
    b_shape = list(b.shape)
    
    while len(a_shape) < len(b_shape):
        a_shape.insert(0, 1)
    while len(b_shape) < len(a_shape):
        b_shape.insert(0, 1)
        
    a_reshaped = a.reshape(a_shape)
    b_reshaped = b.reshape(b_shape)
    
    # 2. 计算目标广播形状
    target_shape = []
    for dim_a, dim_b in zip(a_shape, b_shape):
        if dim_a == dim_b:
            target_shape.append(dim_a)
        elif dim_a == 1:
            target_shape.append(dim_b)
        elif dim_b == 1:
            target_shape.append(dim_a)
        else:
            raise ValueError(f"Operands could not be broadcast together with shapes {a.shape} {b.shape}")
            
    # 3. 手动沿着大小为 1 的轴拉伸数据 (模拟底层广播数据拷贝)
    a_broadcasted = a_reshaped.copy()
    b_broadcasted = b_reshaped.copy()
    
    for axis, (dim_a, dim_b) in enumerate(zip(a_shape, b_shape)):
        if dim_a == 1 and dim_b > 1:
            a_broadcasted = np.repeat(a_broadcasted, dim_b, axis=axis)
        if dim_b == 1 and dim_a > 1:
            b_broadcasted = np.repeat(b_broadcasted, dim_a, axis=axis)
            
    return a_broadcasted, b_broadcasted

## 二、 手动实现矩阵乘法：manual_matmul(x, y)
底层硬核逻辑：
为了强烈感受 $O(N^3)$ 的时间复杂度，严格采用三层 for 循环。第 1 层遍历 X 的行，第 2 层遍历 Y 的列，第 3 层作为“内积传送带”进行对应元素点乘并累加。

def manual_matmul(x, y):
    if x.ndim != 2 or y.ndim != 2:
        raise ValueError("仅支持二维矩阵乘法进行硬核体验")
    if x.shape[1] != y.shape[0]:
        raise ValueError(f"矩阵内侧维度不匹配: {x.shape[1]} vs {y.shape[0]}")
        
    H, W_mid = x.shape
    _, W_out = y.shape
    
    # 预先开辟一整块连续内存空间
    out = np.zeros((H, W_out), dtype=np.float64)
    
    # 三层大循环，严禁使用任何高级向量化技巧
    for i in range(H):          # 遍历左矩阵的每一行
        for j in range(W_out):  # 遍历右矩阵的每一列
            sum_val = 0.0
            for k in range(W_mid): # 纽带轴，进行 C 级别的指针累加
                sum_val += x[i, k] * y[k, j]
            out[i, j] = sum_val
            
    return out

## 三、 手动实现 Softmax（防上溢/下溢）：manual_softmax(x)
底层硬核逻辑：
若直接计算 $e^x$，当 $x > 700$ 时会发生浮点数上溢（变成 inf）。利用数学恒等式 $\text{Softmax}(x) = \text{Softmax}(x - C)$，令 $C = \max(x)$，使指数项最大为 $e^0 = 1$，彻底杜绝溢出。使用 keepdims=True 确保减法时能正确触发表两路广播。

import scipy.special
def manual_softmax(x):
    if x.ndim != 2:
        raise ValueError("假定输入为 (Batch_Size, Classes) 的二维特征图")
        
    # 1. 找到每行（每个样本）的最大值，必须保留维度 (Batch_Size, 1) 才能触发广播减法
    row_max = x.max(axis=1, keepdims=True)
    
    # 2. 防溢出平移 (此时每行最大的数变成了 0，其余全为负数，e^负数 稳定在 0~1 之间)
    x_shifted = x - row_max
    
    # 3. 计算指数项
    exp_x = np.exp(x_shifted)
    
    # 4. 按行求和，同样保留维度 (Batch_Size, 1) 用于最后的广播除法
    sum_exp = exp_x.sum(axis=1, keepdims=True)
    
    return exp_x / sum_exp
# --- 验证模块 ---# 故意填入一个能让普通 softmax 崩溃的极限大数logits = np.array([[1000.0, 1001.0, 999.0], [-10.0, 0.0, 10.0]])res_manual = manual_softmax(logits)res_scipy = scipy.special.softmax(logits, axis=1)assert np.allclose(res_manual, res_scipy), "Softmax 逻辑不一致或发生了溢出！"
print("✅ 模块 3：防溢出 Softmax 验证通过！")

## 四、 手动加载并显示 MNIST 图片
底层硬核逻辑：
由于网络上下载的 MNIST 官方数据集通常是 .gz 格式的原始二进制大端（Big-Endian）文件，我们需要：

   1. 用 gzip 解压字节流。
   2. 用 np.frombuffer 解析魔数（Magic Number）、图片总数和行列数。
   3. 利用指针偏移量跳过文件头（十六进制描述符），将一维的一连串字节流按照 (N, 28, 28) 的 shape 强行重塑（Reshape）转换。

前置条件：请确保本地当前目录下有官方的 train-images-idx3-ubyte.gz 和 train-labels-idx1-ubyte.gz 文件。

import gzip
def load_mnist_raw(images_path, labels_path):
    # 1. 解析图片二进制文件
    with gzip.open(images_path, 'rb') as f:
        # 读取前 16 个字节的头部元数据
        # 包含：魔数 (4字节), 图片数 (4字节), 行数 (4字节), 列数 (4字节)
        header = np.frombuffer(f.read(16), dtype=np.dtype('>i4'))
        magic, num_images, rows, cols = header
        
        # 剩余的数据全部读入，作为单字节无符号整数 (uint8)
        images = np.frombuffer(f.read(), dtype=np.uint8)
        # 强行改变元数据形状，从一维指针块变成三维空间表格
        images = images.reshape(num_images, rows, cols)
        
    # 2. 解析标签二进制文件
    with gzip.open(labels_path, 'rb') as f:
        # 读取前 8 个字节的头部元数据：魔数 (4字节), 标签数 (4字节)
        header = np.frombuffer(f.read(8), dtype=np.dtype('>i4'))
        magic, num_labels = header
        labels = np.frombuffer(f.read(), dtype=np.uint8)
        
    return images, labels

## 五、 手动实现 Minibatch 切片与 Shuffle 逻辑
底层硬核逻辑：
绝不能每次直接对庞大的 X_train 执行整体拷贝和打乱（这会导致极其恐怖的内存搬运开销）。高效的做法是：

   1. 创建一个与数据集等长的一维整型索引数组（Index Array）。
   2. 仅对这个索引数组进行打乱（Shuffle）。
   3. 按照指定的 batch_size 步长切出当前的索引片段，最后利用花式索引（Fancy Indexing）一次性从物理内存中将对应的图片抽取出来。

def minibatch_generator(X, y, batch_size=100, shuffle=True):
    num_samples = X.shape[0]
    
    # 1. 建立虚拟的索引传送带，避免直接搬运高维的大型图像内存块
    indices = np.arange(num_samples)
    
    # 2. 如果开启 shuffle，仅原地打乱这一层一维索引
    if shuffle:
        np.random.shuffle(indices)
        
    # 3. 按照步长切片，滚动输出
    for start_idx in range(0, num_samples, batch_size):
        end_idx = min(start_idx + batch_size, num_samples)
        
        # 抽出当前批次的局部索引（例如 100 个随机数字）
        batch_indices = indices[start_idx:end_idx]
        
        # 4. 利用花式索引（激活 C 级内存拷贝）交付最终数据
        X_batch = X[batch_indices]
        y_batch = y[batch_indices]
        
        yield X_batch, y_batch
